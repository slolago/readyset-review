**Edit 2 — Add viewMode state after existing state declarations (around line 65, after the multi-select state block):**
```typescript
// View mode: 'grid' | 'list', persisted per folder
const viewModeKey = `view-mode-${folderId ?? 'root'}`;
const [viewMode, setViewMode] = useState<'grid' | 'list'>(() => {
  if (typeof window === 'undefined') return 'grid';
  return (localStorage.getItem(viewModeKey) as 'grid' | 'list') ?? 'grid';
});

useEffect(() => {
  localStorage.setItem(viewModeKey, viewMode);
}, [viewModeKey, viewMode]);
```

Note: The lazy initializer reads localStorage once on mount. The useEffect persists changes. When `folderId` changes (user navigates to a subfolder), `viewModeKey` changes, triggering a re-read via a separate effect:
```typescript
useEffect(() => {
  if (typeof window === 'undefined') return;
  const stored = localStorage.getItem(viewModeKey) as 'grid' | 'list' | null;
  if (stored) setViewMode(stored);
}, [viewModeKey]);
```
