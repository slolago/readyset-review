Fix EDIT-01: inline rename must cancel on outside-click (not commit), and only one rename input may be active across the whole file browser at any time — starting rename B on a different card cancels A first.

Purpose: Current behavior loses data surprisingly (users expect clicking away to be cancel, not lose-the-edit) and allows multiple simultaneous renames that each write conflicting PUTs. This fixes both.

Output: A hardened `<InlineRename>` primitive with click-away cancel, plus a `FolderBrowser`-scoped `activeRenameId` singleton that all four rename call sites (FolderCard, FolderListRow, AssetCard, AssetListRow) consume.
</objective>

  interface RenameController {
    activeId: string | null;
    setActiveId: (id: string | null) => void;
  }

  const RenameControllerContext = React.createContext<RenameController>({
    activeId: null,
    setActiveId: () => {},
  });

  function RenameProvider({ children }: { children: React.ReactNode }) {
    const [activeId, setActiveId] = React.useState<string | null>(null);
    const value = React.useMemo(() => ({ activeId, setActiveId }), [activeId]);
    return <RenameControllerContext.Provider value={value}>{children}</RenameControllerContext.Provider>;
  }

  export function useRenameController(): RenameController {
    return React.useContext(RenameControllerContext);
  }
