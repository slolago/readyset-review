Soft-delete convention: docs have `deletedAt: Timestamp` and `deletedBy: string` when trashed. Absence (null/undefined) means live. Filter pattern in this codebase: in-memory `!doc.deletedAt` after a fetch, OR `.where('deletedAt', '==', null)` when it doesn't require a composite index. Prefer in-memory filter to stay index-free (Phase 52 precedent, per CONTEXT.md).

Stack identity: a version group is every asset doc whose `versionGroupId == groupId`. Root may be legacy (missing the field) — always use `fetchGroupMembers` for group reads.

    Step A — Create `src/lib/folders.ts` with a BFS deep-copy helper (file does not exist yet):

    ```ts
    import type { Firestore } from 'firebase-admin/firestore';
    import { Timestamp } from 'firebase-admin/firestore';

    export interface DeepCopyCounts { folders: number; assets: number; }

    /**
     * Deep-copy a folder subtree. Creates a new root folder under dstParentId,
     * then BFS through subfolders, re-parenting each copy under its newly-created
     * parent. Skips soft-deleted folders and assets (SDC alignment).
     *
     * Reuses the `assets/copy` shape inline (simple doc clone + new versionGroupId
     * per stack) — NOT calling the API route, to stay server-side.
     * Returns the new root folder id and counts.
     */
    export async function deepCopyFolder(
      db: Firestore,
      srcFolderId: string,
      dstParentId: string | null,
      projectId: string,
      userId: string,
      overrideName?: string,
    ): Promise<{ newRootId: string; counts: DeepCopyCounts }> {
      const srcRoot = await db.collection('folders').doc(srcFolderId).get();
      if (!srcRoot.exists) throw new Error('Source folder not found');
      const rootData = srcRoot.data() as any;

      // Compute path for destination root
      let dstPath: string[] = [];
      if (dstParentId) {
        const p = await db.collection('folders').doc(dstParentId).get();
        if (p.exists) dstPath = [...((p.data() as any).path || []), dstParentId];
      }

      // Create new root folder
      const newRootRef = db.collection('folders').doc();
      await newRootRef.set({
        name: overrideName ?? rootData.name,
        projectId,
        parentId: dstParentId ?? null,
        path: dstPath,
        createdAt: Timestamp.now(),
      });

      // Map srcFolderId -> newFolderId for re-parenting
      const idMap = new Map<string, string>();
      idMap.set(srcFolderId, newRootRef.id);

      let folderCount = 1;
      let assetCount = 0;

      // BFS queue of src folder ids whose children still need cloning
      const queue: string[] = [srcFolderId];
      while (queue.length) {
        const currentSrc = queue.shift()!;
        const currentDst = idMap.get(currentSrc)!;
        const currentDstDoc = await db.collection('folders').doc(currentDst).get();
        const currentDstPath = (currentDstDoc.data() as any).path || [];

        // Copy assets in this src folder
        const assetsSnap = await db.collection('assets').where('folderId', '==', currentSrc).get();
        const liveAssets = assetsSnap.docs.filter((d) => !(d.data() as any).deletedAt);

        // Group by versionGroupId so each stack gets a fresh newGroupId
        const groups = new Map<string, FirebaseFirestore.QueryDocumentSnapshot[]>();
        for (const d of liveAssets) {
          const data = d.data() as any;
          const gid = data.versionGroupId || d.id;
          if (!groups.has(gid)) groups.set(gid, []);
          groups.get(gid)!.push(d);
        }

        // Write in batches of <=400
        let batch = db.batch();
        let inBatch = 0;
        const flush = async () => { if (inBatch > 0) { await batch.commit(); batch = db.batch(); inBatch = 0; } };

        for (const [, members] of groups) {
          const newGroupId = db.collection('assets').doc().id;
          for (const d of members) {
            const data = d.data() as any;
            const newRef = db.collection('assets').doc();
            const copyData: any = {
              ...data,
              folderId: currentDst,
              versionGroupId: newGroupId,
              createdAt: Timestamp.now(),
              uploadedBy: userId,
            };
            delete copyData.id;
            delete copyData.deletedAt;
            delete copyData.deletedBy;
            batch.set(newRef, copyData);
            inBatch++;
            assetCount++;
            if (inBatch >= 400) await flush();
          }
        }
        await flush();

        // Enqueue live subfolders
        const subSnap = await db.collection('folders').where('parentId', '==', currentSrc).get();
        for (const sub of subSnap.docs) {
          const subData = sub.data() as any;
          if (subData.deletedAt) continue;
          const newSubRef = db.collection('folders').doc();
          await newSubRef.set({
            name: subData.name,
            projectId,
            parentId: currentDst,
            path: [...currentDstPath, currentDst],
            createdAt: Timestamp.now(),
          });
          idMap.set(sub.id, newSubRef.id);
          folderCount++;
          queue.push(sub.id);
        }
      }

      return { newRootId: newRootRef.id, counts: { folders: folderCount, assets: assetCount } };
    }
    ```
